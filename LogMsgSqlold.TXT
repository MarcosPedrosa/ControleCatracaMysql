SELECT AAFHTFUN.CHAPA, AAFHTFUN.DATA, AAFHTFUN.EXTRAAUTORIZADO AS NUMHORAS, (SELECT TOP 1 SUBSTRING(CODSECAO,10,3)  FROM PFHSTSEC  WHERE CODCOLIGADA = 1  AND   CHAPA = AAFHTFUN.CHAPA  AND   DTMUDANCA <=  AAFHTFUN.DATA  ORDER BY DTMUDANCA DESC) AS INTEGRGERENCIAL, (SELECT NOME FROM PFUNC WHERE PFUNC.CHAPA = AAFHTFUN.CHAPA AND PFUNC.CODCOLIGADA = '1')  AS NOME_FUNCIONARIO, (SELECT PFUNCAO.NOME FROM PFUNCAO,PFUNC WHERE PFUNC.CODFUNCAO = PFUNCAO.CODIGO AND PFUNC.CODCOLIGADA = PFUNCAO.CODCOLIGADA AND PFUNC.CHAPA = AAFHTFUN.CHAPA AND PFUNCAO.CODCOLIGADA = '1')  AS NOME_FUNCAO FROM AAFHTFUN  WHERE AAFHTFUN.CODCOLIGADA = '1'  AND   CONVERT(NVARCHAR(10),AAFHTFUN.DATA, 121) >= '2011-09-01'  AND   CONVERT(NVARCHAR(10),AAFHTFUN.DATA, 121) < '2012-02-15 ORDER BY CHAPA, DATA DESC
